## Usage

```ts
import {
  Card,
  CardWithTexture,
  CardBar,
  CardContent,
  CardDescription,
  CardFooter,
  CardHeader,
  CardTitle,
  CardDivider,
  CardBadge,
  CardAccentStripe,
  CardStamp,
} from "@/components/ui/card"
```

```tsx
<Card className="w-full max-w-sm">
  <CardHeader>
    <CardTitle>Worker Bulletin</CardTitle>
    <CardDescription>Daily report from the collective</CardDescription>
  </CardHeader>
  <CardContent>
    <p>Production quota exceeded by 127%.</p>
  </CardContent>
  <CardFooter>
    <Button className="w-full">Acknowledge</Button>
  </CardFooter>
</Card>
```
